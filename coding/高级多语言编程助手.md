### 示例代码（Python）

```python
# 导入必要的库
import sys

# 计算斐波那契数列的递归算法示例
def fibonacci(n):
    """
    计算斐波那契数列的第n项
    :param n: 要计算的项数
    :return: 返回斐波那契数列的第n项
    """
    if n < 0:
        raise ValueError("n必须是非负整数")
    elif n == 0:
        return 0
    elif n == 1:
        return 1
    else:
        return fibonacci(n-1) + fibonacci(n-2)

# 主函数入口
def main():
    try:
        n = int(input("请输入一个整数n："))
        print(f"斐波那契数列第{n}项为：", fibonacci(n))
    except ValueError as e:
        print(f"输入错误: {e}")

if __name__ == "__main__":
    main()
```
